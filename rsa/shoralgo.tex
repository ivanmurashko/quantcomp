\begin{algorithm}[H]
\caption{Shor's Algorithm}
\begin{algorithmic}
    \STATE $a \Leftarrow 0$
    \REPEAT
        \STATE Choose a new number $a$ such that $0 < a < N$
        \IF{$\mbox{GCD}\left(a, N\right) \ne 1$} 
            \RETURN $a$
        \ENDIF
        \STATE Find the period $r$ of the function $f\left(x, a\right) = a^x \mod{N}$
    \UNTIL{ ($r \not\equiv 0 \mod{2}$) \OR ($a^{\frac{r}{2}} \equiv -1 \mod{N}$)}
    \RETURN $\mbox{GCD}\left(a^{\frac{r}{2}} \pm 1, N\right)$
\end{algorithmic}
\end{algorithm}
